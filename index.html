// ══════════════════════════════════════════ // AUTH SYSTEM // ══════════════════════════════════════════ const ADMIN_PHONE = '0987654321'; // Updated Admin Number let CURRENT_USER = null; async function handleLoginSubmit(){ const phone = document.getElementById('loginPhone').value.trim(); if(phone.length < 10){ showToast('❌ Enter valid mobile number', false); return; } showToast('⏳ Checking access…', true); // 1. ADMIN LOGIN CHECK if(phone === ADMIN_PHONE || phone === '9876543211'){ const user = {phone, role:'admin', name:'Admin'}; CURRENT_USER = user; sessionStorage.setItem('akdf_session', JSON.stringify(user)); applyUserSession(user); showToast('✅ Welcome Admin', true); return; } // 2. MEMBER LOGIN CHECK const members = await getCollection('members'); const matched = members.find(m => (m.phone||'').replace(/\D/g,'').slice(-10) === phone.slice(-10)); if(matched){ const reqs = await db.collection('accessRequests').where('phone','==',phone).get(); if(!reqs.empty){ const req = reqs.docs[0].data(); if(req.status === 'approved'){ const user = {phone, role:'member', memberId: matched.id, name: matched.name}; CURRENT_USER = user; sessionStorage.setItem('akdf_session', JSON.stringify(user)); applyUserSession(user); } else if(req.status === 'denied'){ showLoginStep('loginStep3'); } else { document.getElementById('pendingPhone').textContent = `+91 ${phone}`; showLoginStep('loginStep2'); } } else { await db.collection('accessRequests').add({ phone, name: matched.name, memberId: matched.id, status: 'pending', requestedAt: new Date().toISOString() }); document.getElementById('pendingPhone').textContent = `+91 ${phone}`; showLoginStep('loginStep2'); showToast('📨 Access request sent to admin', true); } } else { showToast('❌ Number not registered. Contact admin.', false); } } function applyUserSession(user){ document.getElementById('loginScreen').style.display='none'; const logoutBtn = document.getElementById('logoutBtn'); if(logoutBtn) logoutBtn.style.display = 'block'; if(user.role === 'admin'){ document.body.classList.remove('member-mode'); document.getElementById('adminHeader').style.display='flex'; document.getElementById('memberHeader').style.display='none'; // Admin UI visibility document.getElementById('adminStatCards').style.display=''; document.getElementById('adminActionBtns').style.display='flex'; document.getElementById('adminMemberSearch').style.display=''; document.getElementById('adminQuickBtns').style.display='flex'; document.getElementById('bottomNav').style.display='flex'; updateUI(); pollPendingRequests(); } else { // Member Mode document.body.classList.add('member-mode'); document.getElementById('adminHeader').style.display='none'; document.getElementById('memberHeader').style.display='block'; document.getElementById('memberHeaderAvatar').textContent = ini(user.name); document.getElementById('memberHeaderName').textContent = user.name; document.getElementById('memberHeaderPhone').textContent = `📱 +91 ${user.phone}`; document.getElementById('memberLedgerArea').style.display='block'; document.getElementById('bottomNav').style.display='none'; loadMemberLedger(); } }
